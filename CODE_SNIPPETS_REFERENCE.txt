    getEntryNamesForTeam(teamCode) {
      if (!this.activeSession?.allEntries) return null;
      const entry = this.activeSession.allEntries.find(
        e => e.code.toLowerCase() === teamCode.toLowerCase()
      );
      return entry ? entry.entry : null;
    }

7. channel-mapper.js - autoMap()
═══════════════════════════════════════════════════════════════════════════════

LINES: 39-61
CALLED BY: handleInitiatePairings() to auto-detect channels
RETURNS: mapping object with channel info

ALGORITHM:
    For each team code:
      1. Extract suffix: "Interlake CG" → "CG"
      2. Search guilds for channel: "cg-tournaments"
      3. If found: confidence="auto", store channelId
      4. If not: confidence="unmatched", set to null

CODE:
    async autoMap(teamCodes) {
      const mapping = {};
      for (const code of teamCodes) {
        const suffix = this.extractTeamSuffix(code);
        if (!suffix) {
          mapping[code] = { channelId: null, channelName: null, 
                           confidence: 'unmatched' };
          continue;
        }
        const channel = await this.findChannel(suffix);
        if (channel) {
          mapping[code] = {
            channelId: channel.id,
            channelName: channel.name,
            confidence: 'auto',
          };
        } else {
          mapping[code] = { channelId: null, channelName: null, 
                           confidence: 'unmatched' };
        }
      }
      return mapping;
    }

8. report-builder.js - buildFullReport()
═══════════════════════════════════════════════════════════════════════════════

LINES: 97-112
CALLED BY: _processSinglePairing() to combine embeds
RETURNS: Array of Discord embeds (max 10)

CODE:
    buildFullReport(pairing, opponent, judges) {
      const embeds = [];

      if (pairing) {
        embeds.push(this.buildPairingEmbed(pairing, opponent));
      }

      if (Array.isArray(judges)) {
        for (const judge of judges) {
          if (embeds.length >= 10) break;  // Discord limit
          embeds.push(this.buildJudgeEmbed(judge));
        }
      }

      return embeds;
    }
